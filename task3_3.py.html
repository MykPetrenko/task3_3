9    def cell_method(outer_lb, outer_ub, inner_lb, inner_ub, f, n, m):
10       points_n = np.linspace(outer_lb, outer_ub, n)
11       points_m = np.linspace(inner_lb, inner_ub, m)
12       res = 0
13       for i in range(1, n):
14           for j in range(1, m):
15               res += f((points_n[i] + points_n[i - 1]) / 2, (points_m[j] + points_m[j - 1]) / 2) * \
16                  (points_n[i] - points_n[i - 1]) * (points_m[j] - points_m[j - 1])
17   
18       return res
